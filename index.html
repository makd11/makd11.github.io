Program 5: Login Page with Validations using JavaScript

<html>
  <head>
      <title>Login</title>
  </head>
  <body>
      <h2>Login</h2>
  
      <form onsubmit="return validate()">
          <input type="text" id="email" placeholder="Email"><br><br>
          <input type="password" id="pass" placeholder="Password"><br><br>
          <input type="submit" value="Login">
      </form>
  
      <p id="msg" style="color: red;"></p>
  
      <script>
          function validate() {
              let email = document.getElementById("email").value;
              let pass = document.getElementById("pass").value;
              let msg = document.getElementById("msg");
  
              if (email === "" || pass === "") {
                  msg.textContent = "Please fill all fields";
                  return false;
              }
              if (!email.includes("@")) {
                  msg.textContent = "Invalid email";
                  return false;
              }
  
              return true;
          }
      </script>
  </body>
  </html>

Program 6: Simple Calculator using JavaScript

<html>
  <body>
  <h3>Calculator</h3>
  
  <input id="a" type="number">
  <input id="b" type="number"><br><br>
  
  <button onclick="calc('+')">+</button>
  <button onclick="calc('-')">-</button>
  <button onclick="calc('/')">/</button>
  <button onclick="calc('*')">*</button>
  
  <p id="res"></p>
  
  <script>
      function calc(op) {
          let x = parseFloat(document.getElementById("a").value);
          let y = parseFloat(document.getElementById("b").value);
  
          // Validate inputs
          if (isNaN(x) || isNaN(y)) {
              document.getElementById("res").innerText = "Please enter valid numbers.";
              return;
          }
  
          let r = eval(x + op + y);
          document.getElementById("res").innerText = "Result: " + r;
      }
  </script>
  </body>
  </html>

Program 7: Login Page with Regular Expression Validation

<html>
  <body>
      <h3>Login</h3>
  
      <input id="email" placeholder="Email"><br><br>
      <input id="pass" type="password" placeholder="Password"><br><br>
      <button onclick="check()">Login</button>
  
      <p id="msg" style="color: red;"></p>
  
      <script>
          function check() {
              let e = document.getElementById("email").value;
              let p = document.getElementById("pass").value;
              let msg = document.getElementById("msg");
  
              // Regex for email: something@something.suffix (2-3 letters)
              let em = /^[^@]+@[^@]+\\.[a-z]{2,3}$/i;
  
              // Regex for password: 6 or more characters (letters or numbers)
              let pw = /^[A-Za-z0-9]{6,}$/;
  
              if (!em.test(e)) {
                  msg.textContent = "Invalid email";
                  return;
              }
  
              if (!pw.test(p)) {
                  msg.textContent = "Invalid password";
                  return;
              }
  
              msg.style.color = "green";
              msg.textContent = "Login successful";
          }
      </script>
  </body>
  </html>

Program 8: JavaScript Program using Switch Case

<!DOCTYPE html>
  <html>
  <body>
  
    <input id="d" type="number" placeholder="1-7">
    <button onclick="f()">Show</button>
    <p id="r"></p>
  
    <script>
      function f() {
        let d = +document.getElementById("d").value;
        let r;
  
        switch (d) {
          case 1: r = "Sunday"; break;
          case 2: r = "Monday"; break;
          case 3: r = "Tuesday"; break;
          case 4: r = "Wednesday"; break;
          case 5: r = "Thursday"; break;
          case 6: r = "Friday"; break;
          case 7: r = "Saturday"; break;
          default: r = "Invalid";
        }
  
        document.getElementById("r").innerText = r;
      }
    </script>
  
  </body>
  </html>

Program 10: JavaScript Program Using Built-in Functions

<html>
  <body>

    <p id="res"></p>
  
    <script>
      function reverse() {
        let s = document.getElementById("txt").value;

      }
